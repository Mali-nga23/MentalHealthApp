CREATE VIEW IF NOT EXISTS `mental_health_app`.`VIEW_for_chat` AS SELECT `chat_users`.`userid`, `chat_users`.`username`, `chat_users`.`email`, `chat_users`.`password`, `chat_users`.`avatar`, `chat_users`.`current_session`, `chat_users`.`online`, `appointments`.`User_ID`, `appointments`.`Specialist_ID` FROM `chat_users` JOIN `appointments` ON `chat_users`.`userid` = `appointments`.`Specialist_ID`;

CREATE VIEW IF NOT EXISTS `mental_health_app`.`VIEW_for_chat1` AS SELECT `chat_users`.`userid`, `chat_users`.`username`, `chat_users`.`email`, `chat_users`.`password`, `chat_users`.`avatar`, `chat_users`.`current_session`, `chat_users`.`online`, `appointments`.`User_ID`, `appointments`.`Specialist_ID` FROM `chat_users` JOIN `appointments` ON `chat_users`.`userid` = `appointments`.`User_ID`;
